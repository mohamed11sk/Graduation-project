// WebSocket server

FirebaseData fbdo;

// Gas sensor variables
int gasLevel = 0;
unsigned long previousMillis = 0;
const unsigned long sendInterval = 3000;

// Flag to send IP address only once
bool ipSent = false;

void setup() {
  Serial.begin(115200);

  // Connect to WiFi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }
  Serial.print("Connected with IP: ");
  Serial.println(WiFi.localIP());

  // Initialize Firebase Authentication
  config.api_key = API_KEY;
  auth.user.email = USER_EMAIL;
  auth.user.password = USER_PASSWORD;
  config.database_url = DATABASE_URL;
  config.token_status_callback = tokenStatusCallback; // This will still work because it's provided by the library
  Firebase.begin(&config, &auth);  // Ensure Firebase is initialized here

  Firebase.reconnectWiFi(true);

  // Send IP address to Realtime Database only once
  if (!ipSent) {
    sendToRealtimeDatabase();
    ipSent = true; // Set flag to ensure the IP is only sent once
  }

  // Initialize LED
  pinMode(ledPin1, OUTPUT);
  digitalWrite(ledPin1, LOW);

  // Initialize WebSocket
  webSocket.begin();
  webSocket.onEvent(webSocketEvent);

  // Set up time synchronization
  configTime(2 * 3600, 0, "pool.ntp.org");

  // Initialize I2C
  Wire.begin(8);
  Wire.onReceive(receiveEvent);
}

void loop() {
  webSocket.loop();

  // Gas safety override
  if (gasLevel > 50) {
    digitalWrite(ledPin1, LOW);
  }

  unsigned long currentMillis = millis();
  if (currentMillis - previousMillis >= sendInterval) {
    previousMillis = currentMillis;
    sendToFirebase();
  }
}

// WebSocket event handler
void webSocketEvent(uint8_t num, WStype_t type, uint8_t* payload, size_t length) {
  if (type == WStype_TEXT) {
    String message = String((char*)payload);
    if (message.startsWith("LED1:")) {
      if (gasLevel <= 50) {
        int ledState = message.substring(5).toInt();
        digitalWrite(ledPin1, ledState);
      }

      String response = "Client ID: " + String(num) + " LED1: " + String(digitalRead(ledPin1));
      webSocket.sendTXT(num, response);
    }
  }
}

// I2C receive handler
void receiveEvent(int bytes) {
  if (Wire.available() >= 2) {
    byte highByte = Wire.read();
    byte lowByte = Wire.read();
    gasLevel = (highByte << 8) | lowByte;
    Serial.print("Received Gas Level: ");
    Serial.println(gasLevel);
  }
}

// Firebase Firestore send function
void sendToFirebase() {
  time_t timestamp = Firebase.getCurrentTime();
  String timestampStr = formatTimestamp(timestamp);

  String collectionPath = "EspData";
  FirebaseJson content;
  content.set("fields/GasLevel/doubleValue", String(gasLevel));
  content.set("fields/Timestamp/stringValue", timestampStr);

  String documentId = "doc_" + String(millis());

  if (Firebase.Firestore.createDocument(&fbdo, FIREBASE_PROJECT_ID, "", collectionPath.c_str(), documentId.c_str(), content.raw(), "")) {
    Serial.println("Document Added Successfully to Firestore");
  } else {
    Serial.println("Failed to add document to Firestore: " + fbdo.errorReason());
  }
}

// Firebase Realtime Database send function
void sendToRealtimeDatabase() {
  if (Firebase.ready()) { // Check if Firebase is ready
    String ipAddress = WiFi.localIP().toString();
    String path = "/EspData/ip";  // Use a simple path without invalid tokens

    if (Firebase.RTDB.setString(&fbdo, path, ipAddress)) {
      Serial.println("IP address sent to Realtime Database successfully.");
    } else {
      Serial.println("Failed to send IP address to Realtime Database: " + fbdo.errorReason());
    }
  } else {
    Serial.println("Firebase not ready, retrying...");
  }
}

// Timestamp formatter
String formatTimestamp(time_t timestamp) {
  struct tm *timeinfo;
  timeinfo = localtime(&timestamp);
  char buffer[20];
  strftime(buffer, sizeof(buffer), "%Y-%m-%d %H:%M:%S", timeinfo);
  return String(buffer);
}
